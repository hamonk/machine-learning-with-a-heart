def write_prediction(name, prediction):
    r=pd.DataFrame(test_ids.values, prediction, columns=['patient_id']).reset_index()
    r.columns=['heart_disease_present', 'patient_id']
    stamp=datetime.datetime.strftime(datetime.datetime.now(), '%Y%b%d_%Hh%Mmin%Ss')
    file_name = f'prediction_{name}_{stamp}.csv'
    r[['patient_id', 'heart_disease_present']].to_csv(file_name, index=False)
    print(f'done writing {file_name}')
